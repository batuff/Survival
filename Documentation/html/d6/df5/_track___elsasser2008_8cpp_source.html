331  const double r1) const

335  double arg_integral;
336  double d;
337 
338  if( r1 > r_max )

340 
341  if (r1 <= r_min)

343  else

347

351  arg_integral = 0;
352

354  {
355  arg_integral += ak;

357  }

359  }
360  else

368 double Track_Elsasser2008::normalizedPunctualDose(const double distance) const
369 {
370  double integralStart = max( 0.0, distance - numberOfSigma * SIGMA );
371  double integralStop = min( r_max, distance + numberOfSigma * SIGMA );
372  int nr1 = int( ceil( (integralStop-integralStart)/integrationStep ) );
373  double dr1;
374  if( nr1 > 0 )
375  dr1 = (integralStop-integralStart) / nr1;
376  else
377  return 0;
378  double normalizedDose = 0;
379  double r1 = integralStart;
380  double argument_1 = normalizedDoseIntegralArgument(distance, integralStart);
381  double argument_2;
382 
383  do {
384  r1 += dr1;
385  argument_2 = normalizedDoseIntegralArgument(distance, r1);
386  normalizedDose += (argument_1 + argument_2) * dr1 / 2; // trapezoidal rule
387  argument_1 = argument_2;
388  } while( r1 < integralStop );
