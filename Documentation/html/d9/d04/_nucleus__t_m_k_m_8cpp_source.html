228 void Nucleus_tMKM::getDoseAndSurvival(double &dose,
229  double &doseUncertainty,
230  double &survival,
231  double &survivalUncertainty) const
232 {
233  double sumDose = 0.0, sumLethal = 0.0, lethal=0.0;
234  vector<double> doses, times;
235 
236  for( int dom = 0; dom < numberOfDomains; dom++ )
237  {
238  doses = domains[dom]->getDoses();
239  times = domains[dom]->getTimes();
240  if (times.size()!=0) {
241  domains[dom]->getDose(dose);
242  lethal = alpha_d*dose + beta_d*dose*dose;
243 
244  for (unsigned long int i=0; i<times.size()-1; ++i)
245  for (unsigned long int j=i+1; j<times.size(); ++j)
246  lethal -= 2*beta_d * doses[i]*doses[j] * ( 1-exp(-cellLine.getAC()*(times[j]-times[i])) );
247  }
248  else {
249  dose=0.0;
250  lethal=0.0;
251  }
252  sumDose += dose;
253  sumLethal += lethal;
254  }
255  dose = sumDose / numberOfDomains; // average dose.
256  survival = exp(-sumLethal);
257  doseUncertainty = -1;
258  survivalUncertainty = -1;
259 }
